# === 4. Preflight ===
python3 -c "
import sys
sys.path.insert(0, 'parameter-golf')
with open('parameter-golf/train_gpt_v2.py') as f:
    code = f.read()
checks = {
    'GRAD_ACCUM_STEPS env var': 'GRAD_ACCUM_STEPS' in code,
    'PREQUANT_TTT_ENABLED': 'PREQUANT_TTT_ENABLED' in code,
    'prequant_ttt function': 'prequant_ttt' in code,
    'weights_only=False fix': 'weights_only=False' in code,
    'EMA_DECAY env var': 'EMA_DECAY' in code,
    'GATED_ATTN env var': 'GATED_ATTN' in code,
}
all_ok = all(checks.values())
for name, ok in checks.items():
    print(f'  {\"✓\" if ok else \"✗\"} {name}')
print(f'\nPreflight: {\"PASS\" if all_ok else \"FAIL\"} ({sum(checks.values())}/{len(checks)} checks)')
if not all_ok:
    print('ERROR: Missing code. Did you pull the latest James-experiment branch?')
    sys.exit(1)
"
