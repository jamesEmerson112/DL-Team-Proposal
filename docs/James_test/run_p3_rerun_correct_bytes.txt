python3 -c "
from huggingface_hub import hf_hub_download
import shutil, os

cached = hf_hub_download(
    repo_id='romeerp/parameter-golf-caseops-v1',
    filename='fineweb_val_bytes_000000.bin',
    subfolder='datasets/datasets/fineweb10B_sp8192_lossless_caps_caseops_v1_reserved',
    repo_type='dataset',
)
dest = 'data/datasets/fineweb10B_sp8192_lossless_caps_caseops_v1_reserved/fineweb_val_bytes_000000.bin'
os.makedirs(os.path.dirname(dest), exist_ok=True)
try:
    os.link(os.path.realpath(cached), dest)
except OSError:
    shutil.copy2(os.path.realpath(cached), dest)
print(f'Sidecar file saved to: {dest}')
print(f'Size: {os.path.getsize(dest):,} bytes')
"
