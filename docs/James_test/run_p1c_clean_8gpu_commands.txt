ls -la data/datasets/ | grep fineweb10B_sp8192

# a symlink. If you see a symlink, rm and re-download.

# =============================================================================
# STEP 2: Sanity-check the tokenizer is canonical (NOT CaseOps)
# =============================================================================

python3 -c "
import sentencepiece as spm
sp = spm.SentencePieceProcessor()
sp.Load('./data/tokenizers/fineweb_8192_bpe.model')
print('vocab_size:', sp.vocab_size())

# CaseOps would show pieces like '<ᴀ>' or '◌' in the low-ID range.
"

# Use the new fingerprint helper to confirm:

# =============================================================================
# STEP 3: Run P1c — seed 42 (validation pass)
# =============================================================================
# Why one seed first: if size > 16 MB or BPB looks anomalous, fix the config
# before paying for two more seeds.

cd /workspace/DL-Team-Proposal/parameter-golf  # adjust if your pod path differs
source ../runs/configs/p4_clip12.env
torchrun --standalone --nproc_per_node=8 train_gpt.py

# =============================================================================
# STEP 4: Validate seed 42 result before running more seeds
# =============================================================================

# 4a. Confirm the byte source is correct:
grep "tokenizer_fingerprint:" logs/p4_clip12.txt
grep "byte_source:" logs/p4_clip12.txt
# Expected:
#   tokenizer_fingerprint: standard
#   byte_source: lut
# If you see "caseops" or "sidecar", abort — the fix didn't take effect.

# 4b. Confirm the token/byte ratio is in the standard SP8192 range:
python3 -c "
import math, re
with open('logs/p4_clip12.txt') as f:
    text = f.read()
# Last quantized_ttt_phased line:
m = re.findall(r'quantized_ttt_phased val_loss:([\d.]+) val_bpb:([\d.]+)', text)
if not m:
    print('ERROR: no TTT eval line in log — run probably crashed.'); raise SystemExit(1)
val_loss, val_bpb = map(float, m[-1])
ratio = val_bpb * math.log(2) / val_loss
print(f'val_loss={val_loss:.4f}  val_bpb={val_bpb:.4f}  ratio={ratio:.4f}')
print(f'Standard SP8192 expected ratio: ~0.268')
if abs(ratio - 0.268) < 0.005:
    print('OK — ratio is canonical, byte counting is correct.')
else:
    print(f'WARNING — ratio {ratio:.4f} is OFF (expected ~0.268). Investigate.')
"

# 4c. Confirm under-budget:
ls -la final_model.int6.ptz
# Expected: < 16,000,000 bytes (16 MB).
# If over: try EMBED_BITS=6 (~-1 MB, +0.013 BPB cost) before next seed.

# 4d. Confirm the gated attention actually fired:
grep "gated_attn_enabled:" logs/p4_clip12.txt
# Expected: gated_attn_enabled: True

# Decision gate:
#   - Budget OK + ratio ~0.268 + BPB < 1.090: proceed to STEP 5 (more seeds).
#   - Budget OVER: tighten EMBED_BITS or MATRIX_CLIP_SIGMAS, rerun seed 42.
#   - Ratio off or BPB > 1.10: STOP — investigate config or pod state.

# =============================================================================
# STEP 5: Run seeds 1337 and 2025 (only if seed 42 passed STEP 4)
# =============================================================================

# Save seed 42 log so it isn't overwritten:
mv logs/p4_clip12.txt logs/p4_clip12_seed42.txt

SEED=1337 torchrun --standalone --nproc_per_node=8 train_gpt.py
mv logs/p4_clip12.txt logs/p4_clip12_seed1337.txt

SEED=2025 torchrun --standalone --nproc_per_node=8 train_gpt.py
mv logs/p4_clip12.txt logs/p4_clip12_seed2025.txt

# =============================================================================
# STEP 6: Compute 3-seed mean and compare to C6
# =============================================================================

python3 -c "
import re, statistics
seeds = [42, 1337, 2025]
bpbs = []
for s in seeds:
    with open(f'logs/p4_clip12_seed{s}.txt') as f:
        text = f.read()
    m = re.findall(r'quantized_ttt_phased val_loss:[\d.]+ val_bpb:([\d.]+)', text)
    bpbs.append(float(m[-1]))
    print(f'seed {s}: val_bpb = {bpbs[-1]:.4f}')
mean = statistics.mean(bpbs)
std = statistics.pstdev(bpbs)
print(f'Mean: {mean:.4f} (std {std:.4f})')
print(f'C6 reference: 1.0805 (target to beat)')
delta = mean - 1.0805
if delta < 0:
    print(f'P1c BEATS C6 by {-delta:.4f} BPB')
else:
    print(f'P1c WORSE than C6 by {delta:.4f} BPB')
"
