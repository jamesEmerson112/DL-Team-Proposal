# Int6+brotli should save ~25% on artifact size.

# === 1. Setup (skip if already done) ===
git clone https://github.com/jamesEmerson112/DL-Team-Proposal.git
cd DL-Team-Proposal
git fetch origin
git checkout James-experiment

# === 2. Install deps (skip if already done) ===

python -m pip install --upgrade pip

pip install huggingface_hub brotli

python -c "import torch; print(f'PyTorch version: {torch.__version__}')"

rm -f data/manifest.json

cd ..

# === 4. Run int6 sweep (3 runs, ~30 min) ===
bash runs/run_int6_sweep_2gpu.sh
