# =============================================================================
# Paper #16 (LR Warmup) + Paper #5 (Structured FFN) — 2×H100 RunPod
# =============================================================================
#
# Tests two untested papers from the survey against C6 baseline:
#   Paper #16: "Why Warmup the Learning Rate?" (NeurIPS 2024)
#     - Rank 1 skips LR warmup entirely. We test adding real warmup.
#   Paper #5:  "Building on Efficient Foundations: Structured FFN" (NeurIPS 2024)
#     - Replace dense FFN with low-rank + block-diagonal matrices.
#     - Fewer params → more steps in 10 min → potentially better BPB.
#
# 6 runs total (~78 min):
#   R0: C6 baseline (control)           — verify ~1.1622 BPB
#   R1: LR Warmup 2%                    — ~20 steps ramp
#   R2: LR Warmup 5%                    — ~50 steps ramp
#   R3: LR Warmup 10%                   — ~100 steps ramp
#   R4: Structured FFN r=0.5 b=4        — 56% fewer MLP params
#   R5: Structured FFN r=0.75 b=8       — 41% fewer MLP params
#
# Estimated cost: ~$4 for 2xH100 pod (~1.5 hours)
# =============================================================================

# === 1. Setup (new RunPod 2xH100) ===
git clone https://github.com/jamesEmerson112/DL-Team-Proposal.git
cd DL-Team-Proposal
git fetch origin
git checkout James-experiment-2
git pull origin James-experiment-2

# === 2. Install deps ===
cd parameter-golf
python -m pip install --upgrade pip
pip install -r requirements.txt
pip install huggingface_hub brotli
pip install --upgrade torch
python -c "import torch; print(f'PyTorch version: {torch.__version__}')"

# Install FlashAttention-3 (required for V2 — NOT the same as flash-attn pip package)
# Pre-built wheel for H100 (sm90). Check CUDA version first:

python3 -c "import torch; print(f'CUDA: {torch.version.cuda}')"
# CUDA 13.0 wheel:
pip install --no-cache-dir \
  "https://download.pytorch.org/whl/cu130/flash_attn_3-3.0.0-cp39-abi3-manylinux_2_28_x86_64.whl"

# === 3. Download SP8192 dataset (~5-10 min) ===
rm -f data/manifest.json
MATCHED_FINEWEB_REPO_ID=kevclark/parameter-golf python3 data/cached_challenge_fineweb.py --variant sp8192 --train-shards 80

# === 4. Preflight check ===
python3 -c "

print('Preflight PASSED')
"

cd ..  # back to DL-Team-Proposal root

# =============================================================================

cd parameter-golf
bash ../runs/run_v2_paper16_paper5_2gpu.sh
