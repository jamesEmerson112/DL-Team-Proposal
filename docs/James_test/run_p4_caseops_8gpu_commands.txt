# 5c. Confirm token/byte ratio is CaseOps-sidecar (~0.3167):
python3 -c "
import math, re
with open('logs/p4_caseops.txt') as f:
    text = f.read()
m = re.findall(r'quantized_ttt_phased val_loss:([\d.]+) val_bpb:([\d.]+)', text)
if not m:
    print('ERROR: no TTT eval line in log'); raise SystemExit(1)
val_loss, val_bpb = map(float, m[-1])
ratio = val_bpb * math.log(2) / val_loss
print(f'val_loss={val_loss:.4f}  val_bpb={val_bpb:.4f}  ratio={ratio:.4f}')
print(f'CaseOps sidecar expected ratio: ~0.317')
if abs(ratio - 0.317) < 0.010:
    print('OK - ratio matches CaseOps sidecar byte counting.')
elif abs(ratio - 0.268) < 0.010:
    print('WARNING - ratio matches STANDARD, not CaseOps! Check CASEOPS_ENABLED.')
elif abs(ratio - 0.291) < 0.010:
    print('WARNING - ratio matches CaseOps LUT (inflated)! Sidecar not loaded.')
else:
    print(f'UNEXPECTED ratio {ratio:.4f} - investigate.')
"

# =============================================================================
# STEP 7: Compute 3-seed mean
# =============================================================================

python3 -c "
import re, statistics
seeds = [42, 1337, 2025]
bpbs = []
for s in seeds:
    with open(f'logs/p4_caseops_seed{s}.txt') as f:
        text = f.read()
    m = re.findall(r'quantized_ttt_phased val_loss:[\d.]+ val_bpb:([\d.]+)', text)
    bpbs.append(float(m[-1]))
    print(f'seed {s}: val_bpb = {bpbs[-1]:.4f}')
mean = statistics.mean(bpbs)
std = statistics.pstdev(bpbs)
print(f'Mean: {mean:.4f} (std {std:.4f})')
print(f'C6 reference (standard): 1.0805')
delta = mean - 1.0805
if delta < 0:
    print(f'P4_caseops BEATS C6 by {-delta:.4f} BPB')
else:
    print(f'P4_caseops WORSE than C6 by {delta:.4f} BPB')
"
